<script>
var map,sesiones=[],capas={},selId=null,filtro='all',poll=null,linkBoxId=null;
var COLS=['#34d399','#64a0ff','#f59e0b','#ef4444','#a78bfa','#ec4899','#06b6d4','#84cc16'];

function initMap(){
  map=L.map('map').setView([-40.15,-71.35],13);
  L.tileLayer('https://{s}.basemaps.cartocdn.com/dark_all/{z}/{x}/{y}{r}.png',{attribution:'© OSM © CARTO',maxZoom:19}).addTo(map);
}

async function crearSesion(){
  var b=document.getElementById('newBtn');
  b.textContent='Creando...';b.disabled=true;
  try{
    var r=await fetch('/api/session',{method:'POST'});
    var d=await r.json();
    linkBoxId=d.id;
    await cargarTodo();
  }catch(e){
    alert('Error. Si es la primera vez, el servidor tarda ~30 seg en despertar. Probá de nuevo.');
  }
  b.textContent='+ Nueva solicitud';b.disabled=false;
}

async function cargarTodo(){
  try{
    var r=await fetch('/api/sessions');
    sesiones=await r.json();
    renderLista();
    actualizarMapa();
    document.getElementById('cnt').textContent=sesiones.length;
    if(selId){var s=sesiones.find(function(x){return x.id===selId});if(s)verDetalle(selId);}
  }catch(e){}
}

function renderLista(){
  var c=document.getElementById('slist');
  var em=document.getElementById('emptyMsg');
  var f=sesiones.filter(function(s){
    if(filtro==='live')return s.active;
    if(filtro==='stopped')return s.stopped;
    if(filtro==='waiting')return !s.active&&!s.stopped;
    return true;
  });
  if(f.length===0){c.innerHTML='';c.appendChild(em);em.style.display='flex';return;}
  em.style.display='none';
  var html='';

  // Show link box for newly created session
  if(linkBoxId){
    var lnk=window.location.origin+'/s/'+linkBoxId;
    html+='<div class="link-box">';
    html+='<div class="link-box-title">📍 NUEVA SOLICITUD CREADA</div>';
    html+='<div class="link-box-url" id="linkUrl">'+lnk+'</div>';
    html+='<div class="link-box-btns">';
    html+='<button class="btn btn-green" onclick="copiarLink()">📋 Copiar link</button>';
    html+='<button class="btn btn-outline" onclick="cerrarLinkBox()">✕ Cerrar</button>';
    html+='</div></div>';
  }

  for(var i=0;i<f.length;i++){
    var s=f[i];
    var col=COLS[i%COLS.length];
    var badge;
    if(s.stopped)badge='<span class="badge b-stop">FINALIZADA</span>';
    else if(s.active)badge='<span class="badge b-live"><span class="dot"></span>EN VIVO</span>';
    else if(s.termsAcceptedAt)badge='<span class="badge b-terms">T&C OK</span>';
    else if(s.openedAt)badge='<span class="badge b-wait">ABIERTO</span>';
    else badge='<span class="badge b-wait">ESPERANDO</span>';

    var time=new Date(s.createdAt).toLocaleString('es-AR',{hour:'2-digit',minute:'2-digit',day:'2-digit',month:'2-digit'});
    var coord=s.lastPoint?s.lastPoint.lat.toFixed(4)+', '+s.lastPoint.lng.toFixed(4):'Sin ubicación';
    html+='<div class="scard'+(selId===s.id?' sel':'')+'" onclick="seleccionar(\''+s.id+'\')" style="border-left:3px solid '+col+'">';
    html+='<div style="display:flex;justify-content:space-between;align-items:center"><span class="sid">#'+s.id+'</span>'+badge+'</div>';
    html+='<div class="smeta">📅 '+time+' · 📍 '+s.pointCount+' pts'+(s.hasPhoto?' · 📸':'')+'<br><span style="font-family:JetBrains Mono;font-size:10px;color:#3a4a50">'+coord+'</span></div>';
    html+='</div>';
  }
  c.innerHTML=html;
}

function copiarLink(){
  var el=document.getElementById('linkUrl');
  if(el){
    navigator.clipboard.writeText(el.textContent).then(function(){
      el.style.background='rgba(45,138,110,.2)';
      setTimeout(function(){el.style.background='rgba(0,0,0,.3)';},1500);
    });
  }
}

function cerrarLinkBox(){linkBoxId=null;renderLista();}

function actualizarMapa(){
  for(var i=0;i<sesiones.length;i++){
    var s=sesiones[i];
    var col=COLS[i%COLS.length];
    if(!capas[s.id])capas[s.id]={mk:null,pulse:null,pl:null,start:null,cnt:0};
    var L2=capas[s.id];
    if(s.pointCount>0&&s.lastPoint){
      if(!L2.mk){
        L2.mk=L.circleMarker([s.lastPoint.lat,s.lastPoint.lng],{radius:s.active?10:7,fillColor:col,fillOpacity:1,color:'#fff',weight:2}).addTo(map);
        L2.mk.bindPopup('<b>#'+s.id+'</b><br>'+(s.active?'🟢 En vivo':'⬛ Fin')+'<br>📍 '+s.pointCount+' pts');
        L2.mk.on('click',function(id){return function(){seleccionar(id);};}(s.id));
      }else{
        L2.mk.setLatLng([s.lastPoint.lat,s.lastPoint.lng]);
        L2.mk.setStyle({radius:s.active?10:7,fillColor:col});
        L2.mk.setPopupContent('<b>#'+s.id+'</b><br>'+(s.active?'🟢 En vivo':'⬛ Fin')+'<br>📍 '+s.pointCount+' pts');
      }
      if(s.active&&!L2.pulse){
        L2.pulse=L.circleMarker([s.lastPoint.lat,s.lastPoint.lng],{radius:18,fillColor:col,fillOpacity:.12,color:col,weight:1,opacity:.2}).addTo(map);
      }else if(s.active&&L2.pulse){L2.pulse.setLatLng([s.lastPoint.lat,s.lastPoint.lng]);}
      else if(!s.active&&L2.pulse){map.removeLayer(L2.pulse);L2.pulse=null;}
    }
  }
}

async function seleccionar(id){
  selId=id;renderLista();
  try{
    var r=await fetch('/api/session/'+id);
    var d=await r.json();
    var idx=sesiones.findIndex(function(x){return x.id===id;});
    var col=COLS[idx%COLS.length];
    var L2=capas[id];
    if(L2&&L2.pl){map.removeLayer(L2.pl);L2.pl=null;}
    if(L2&&L2.start){map.removeLayer(L2.start);L2.start=null;}
    if(d.points.length>0){
      var ll=d.points.map(function(p){return[p.lat,p.lng];});
      L2.pl=L.polyline(ll,{color:col,weight:3,opacity:.7}).addTo(map);
      L2.start=L.circleMarker([d.points[0].lat,d.points[0].lng],{radius:5,fillColor:'#fff',fillOpacity:1,color:col,weight:2}).addTo(map);
      map.fitBounds(L2.pl.getBounds(),{padding:[50,50]});
    }
    mostrarDetalle(d,id);
  }catch(e){}
}

function mostrarDetalle(d,id){
  document.getElementById('dpanel').classList.remove('hide');
  document.getElementById('did').textContent='#'+id;
  var bg=document.getElementById('dbadge');
  if(d.stopped){bg.className='badge b-stop';bg.textContent='FINALIZADA';}
  else if(d.active){bg.className='badge b-live';bg.innerHTML='<span class="dot"></span>EN VIVO';}
  else{bg.className='badge b-wait';bg.textContent='ESPERANDO';}

  var h='';
  if(d.pointCount>0){
    var dist=0;
    for(var i=1;i<d.points.length;i++)dist+=hav(d.points[i-1],d.points[i]);
    var ds=dist>1000?(dist/1000).toFixed(1)+' km':Math.round(dist)+' m';
    var ts='—';
    if(d.points.length>=2){
      var mins=Math.round((new Date(d.points[d.points.length-1].timestamp)-new Date(d.points[0].timestamp))/60000);
      ts=mins>=60?Math.floor(mins/60)+'h '+(mins%60)+'m':mins+'m';
    }
    h+='<div class="statrow"><div class="statbox"><div class="statv">'+d.pointCount+'</div><div class="statl">PUNTOS</div></div><div class="statbox"><div class="statv">'+ds+'</div><div class="statl">DISTANCIA</div></div><div class="statbox"><div class="statv">'+ts+'</div><div class="statl">TIEMPO</div></div></div>';
    var last=d.points[d.points.length-1];
    h+='<div class="coordbox">📍 '+last.lat.toFixed(6)+', '+last.lng.toFixed(6)+'</div>';
    h+='<a class="gmaps" href="https://www.google.com/maps?q='+last.lat+','+last.lng+'" target="_blank">🗺 Google Maps →</a>';
  }
  if(d.hasPhoto){
    (async function(){
      try{var r=await fetch('/api/photo/'+id);var p=await r.json();
      var img=document.createElement('img');img.src=p.photo;img.className='phthumb';img.onclick=function(){window.open(p.photo)};
      var ph=document.getElementById('photoArea');if(ph)ph.appendChild(img);
      }catch(e){}
    })();
    h+='<div id="photoArea" style="font-size:11px;color:#5a6a70">📸 Foto:</div>';
  }
  h+='<div style="font-size:10px;color:#5a6a70;font-weight:600;margin:12px 0 6px">REGISTRO:</div>';
  function ev(ic,tx,tm){return '<div class="erow"><span>'+ic+'</span><span style="color:#a0a8ac">'+tx+'</span><span class="etime">'+(tm||'')+'</span></div>';}
  h+=ev('🔗','Creada',new Date(d.createdAt).toLocaleTimeString('es-AR'));
  if(d.openedAt)h+=ev('📱','Link abierto',new Date(d.openedAt).toLocaleTimeString('es-AR'));
  if(d.termsAcceptedAt)h+=ev('✅','T&C aceptados',new Date(d.termsAcceptedAt).toLocaleTimeString('es-AR'));
  if(d.pointCount>0)h+=ev('📍',d.pointCount+' puntos',new Date(d.points[0].timestamp).toLocaleTimeString('es-AR'));
  if(d.hasPhoto)h+=ev('📸','Foto','');
  if(d.stopped)h+=ev('⬛','Finalizada','');

  if(d.active)h+='<button class="btn btn-red" style="width:100%;margin-top:10px" onclick="pararSesion(\''+id+'\')">⬛ Detener</button>';
  h+='<button class="btn btn-outline" style="width:100%;margin-top:6px" onclick="navigator.clipboard.writeText(window.location.origin+\'/s/'+id+'\')">📋 Copiar link</button>';

  document.getElementById('dbody').innerHTML=h;
}

function cerrarDetalle(){
  document.getElementById('dpanel').classList.add('hide');
  selId=null;renderLista();
  for(var id in capas){
    if(capas[id].pl){map.removeLayer(capas[id].pl);capas[id].pl=null;}
    if(capas[id].start){map.removeLayer(capas[id].start);capas[id].start=null;}
  }
}

async function pararSesion(id){
  if(!confirm('¿Detener?'))return;
  await fetch('/api/stop/'+id,{method:'POST'});
  cargarTodo();
}

function filtrar(f,btn){
  filtro=f;
  document.querySelectorAll('.tab').forEach(function(t){t.classList.remove('active');});
  btn.classList.add('active');
  renderLista();
}

function hav(a,b){var R=6371000,r=function(d){return d*Math.PI/180;},dLat=r(b.lat-a.lat),dLng=r(b.lng-a.lng),x=Math.sin(dLat/2)*Math.sin(dLat/2)+Math.cos(r(a.lat))*Math.cos(r(b.lat))*Math.sin(dLng/2)*Math.sin(dLng/2);return R*2*Math.atan2(Math.sqrt(x),Math.sqrt(1-x));}

initMap();
cargarTodo();
poll=setInterval(cargarTodo,4000);
</script>
